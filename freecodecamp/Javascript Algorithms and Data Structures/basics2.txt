# object in JavaScript

var cat = {
	"name": "Whiskers",
	"legs": 4,
	"tails": 2,
	"pets": ["cat", "dog"],
	5: "five",
	something: "nothing"
};

* Access object using .
eg. alert(cat.name);

* Access object using []
eg. alert(cat["name"]);

* Updating object using . or []
cat.name = 'Loo';
cat["legs"] = 2;
